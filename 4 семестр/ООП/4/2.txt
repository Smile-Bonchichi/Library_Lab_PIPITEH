using namespace std;

class Fraction {
	double chislo;
public:
	Fraction(double x) {
		this->chislo = x;
	}
	Fraction operator+ (Fraction a) {
		return (this->chislo + a.chislo);
	}
	Fraction operator- (Fraction a) {
		return (this->chislo - a.chislo);
	}
	Fraction operator* (Fraction a) {
		return (this->chislo * a.chislo);
	}
	Fraction operator/ (Fraction a) {
		return (this->chislo / a.chislo);
	}
	Fraction operator++ (int) {
		return (this->chislo += 0.1);
	}
	Fraction operator-- (int) {
		return (this->chislo -= 0.1);
	}
	Fraction operator++ () {
		return (++this->chislo);
	}
	Fraction operator-- () {
		return (--this->chislo);
	}
	void output() {
		cout << chislo << endl;
	}
};
