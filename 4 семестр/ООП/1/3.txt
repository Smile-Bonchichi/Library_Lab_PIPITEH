bool check_number(str s) {
	if (atoi(s.c_str()) || s == "0")
		return true;
	else return false;
}

struct stack_info {
	int item;
	stack_info* next;
};

class Stack_class {
	stack_info* S;
	int count;
private:
	void d(int& k1, int& k2) {
		stack_info* pv = S;
		while (pv != NULL) {
			if (pv->item % 2 == 0)
				k1++;
			else k2++;
			pv = pv->next;
		}
	}
public:
	Stack_class() {
		S = NULL;
		count = 0;
	}
	void push(int s) {
		stack_info* pv = new stack_info;
		pv->item = s;
		pv->next = S;
		S = pv;
		count++;
	}
	void pop() {
		if (count == 0)
			cout << "Стек пуст!!!" << endl << "Нечего удалять!!!" << endl;
		else {
			stack_info* pv = S;
			S = S->next;
			cout << "Элемент " << pv->item << " удален!!!" << endl;
			delete pv;
			count--;
		}
	}
	void output() {
		if (count == 0)
			cout << "Стек пуст!!!" << endl << "Нечего показывать!!!" << endl;
		else {
			stack_info* pv = S;
			int i = count;
			cout << "_____________________" << endl;
			cout << "| Элементы | Индекс |" << endl;
			while (pv != NULL) {
				cout << "| " << setw(8) << pv->item << " | " << setw(6) << i << " |" << endl;
				pv = pv->next;
				i--;
			}
		}
	}
	int getCount() {
		return count;
	}
	void divide() {
		if (count == 0)
			cout << "Стек пуст!!!" << endl << "Нечего разделять!!!" << endl;
		else {
			int countE = 0, countO = 0;
			int i = 0, j = 0, min, max;
			d(countE, countO);
			int* arrEven = new int[countE + 1];
			int* arrOdd = new int[countO + 1];
			stack_info* pv = S;
			while (pv != NULL) {
				if (pv->item % 2 == 0) {
					arrEven[i] = pv->item;
					i++;
				} else {
					arrOdd[j] = pv->item;
					j++;
				}
				pv = pv->next;
			}
			cout << "_____________________" << endl;
			cout << "| Четные | Нечетные |" << endl;
			if (countE > countO) {
				max = countE;
				min = countO;
			}
			else {
				min = countE;
				max = countO;
			}
			for (int i = 0; i < min; i++)
				cout << "| " << setw(6) << arrEven[i] << " | " << setw(8) << arrOdd[i] << " |" << endl;
			if (max == countE)
				for (int i = min; i < max; i++)
					cout << "| " << setw(6) << arrEven[i] << endl;
			else
				for (int i = min; i < max; i++)
					cout << "| " << setw(9) << " | " << setw(8) << arrOdd[i] << " |" << endl;
			delete[]arrEven;
			delete[]arrOdd;
		}
	}
	~Stack_class() {
		delete S;
		count = NULL;
	}
};
