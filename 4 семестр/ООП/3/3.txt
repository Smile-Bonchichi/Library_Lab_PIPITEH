#include <iostream>
#include <ctime>
#include <fstream>
#include <string>

#include <time.h>
#include <iomanip>

#define ll long long
#define str string

using namespace std;

struct speed {
	int plain = 0;
	int mountain = 0;
	int desert = 0;
};

class Tatra;
class Kamaz;

class Kamaz {
	str name; speed S;
public:
	void init(str name, speed s) { this->name = name; this->S = s; }
	void output() {
		cout << "Название Камаза: " << name << endl;
		cout << "Скорость в равнинах: " << S.plain << endl;
		cout << "Скорость в горах: " << S.mountain << endl;
		cout << "Скорость в пустыне: " << S.desert << endl;
	}
	friend class Tatra;
};

class Tatra {
	str name; speed S;
public:
	void init(str name, speed s) { this->name = name; this->S = s; }
	void output() {
		cout << "Название Татры: " << name << endl;
		cout << "Скорость в равнинах: " << S.plain << endl;
		cout << "Скорость в горах: " << S.mountain << endl;
		cout << "Скорость в пустыне: " << S.desert << endl;
	}
	int FrCreater(Kamaz& K, int choice) {
		if (choice == 1) {
			if (K.S.plain > this->S.plain) return 1;
			else if (K.S.plain == this->S.plain) return 0;
			else if (K.S.plain < this->S.plain) return -1;
		}
		else if (choice == 2) {
			if (K.S.mountain > this->S.mountain) return 1;
			else if (K.S.mountain == this->S.mountain) return 0;
			else if (K.S.mountain < this->S.mountain) return -1;
		}
		else if (choice == 3) {
			if (K.S.desert > this->S.desert) return 1;
			else if (K.S.desert == this->S.desert) return 0;
			else if (K.S.desert < this->S.desert) return -1;
		}
	}
};

int main() {
	SetConsoleCP(1251);
	SetConsoleOutputCP(1251);
	
	Kamaz K1; speed K1S;
	K1S.plain = 250; K1S.mountain = 230; K1S.desert = 240;

	Tatra T1; speed T1S;
	T1S.plain = 230; T1S.mountain = 250; T1S.desert = 240;

	K1.init("K1", K1S); T1.init("T1", T1S);

	K1.output(); cout << endl << endl; T1.output();

	cout << endl << endl;
	if (T1.FrCreater(K1, 1) == 1) cout << "Камаз быстрее Татры на равнинах!!!" << endl;
	else if (T1.FrCreater(K1, 1) == 0) cout << "Камаз и Татра равны по скорости на равнинах!!!" << endl;
	else cout << "Татра быстрее Kамаза на равнинах!!!" << endl;

	cout << endl;
	if (T1.FrCreater(K1, 2) == 1) cout << "Камаз быстрее Татры в горах!!!" << endl;
	else if (T1.FrCreater(K1, 2) == 0) cout << "Камаз и Татра равны по скорости в горах!!!" << endl;
	else cout << "Татра быстрее Kамаза в горах!!!" << endl;

	cout << endl;
	if (T1.FrCreater(K1, 3) == 1) cout << "Камаз быстрее Татры в пустыне!!!" << endl;
	else if (T1.FrCreater(K1, 3) == 0) cout << "Камаз и Татра равны по скорости в пустыне!!!" << endl;
	else cout << "Татра быстрее Kамаза в пустыне!!!" << endl;

	return 0;
}
