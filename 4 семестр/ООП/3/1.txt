using namespace std;

struct speed {
	int plain = 0;
	int mountain = 0;
	int desert = 0;
};

class Kamaz {
	str name; speed S;
public:
	void init(str name, speed s) { this->name = name; this->S = s; }
	int getSpeed(int choice) {
		if (choice == 1) return this->S.plain;
		else if (choice == 2) return this->S.mountain;
		else return this->S.desert;
	}
	void output() {
		cout << "Название Камаза: " << name << endl;
		cout << "Скорость в равнинах: " << S.plain << endl;
		cout << "Скорость в горах: " << S.mountain << endl;
		cout << "Скорость в пустыне: " << S.desert << endl;
	}
};

class Tatra {
	str name; speed S;
public:
	void init(str name, speed s) { this->name = name; this->S = s; }
	int getSpeed(int choice) {
		if (choice == 1) return this->S.plain;
		else if (choice == 2) return this->S.mountain;
		else return this->S.desert;
	}
	void output() {
		cout << "Название Татры: " << name << endl;
		cout << "Скорость в равнинах: " << S.plain << endl;
		cout << "Скорость в горах: " << S.mountain << endl;
		cout << "Скорость в пустыне: " << S.desert << endl;
	}
};

int FrCreater(int K, int T) {
	if (K > T) return 1;
	else if (K == T) return 0;
	else if (K < T) return -1;
	return 0;
}

int main() {
	SetConsoleCP(1251);
	SetConsoleOutputCP(1251);

	Kamaz K1; speed K1S;
	K1S.plain = 250; K1S.mountain = 230; K1S.desert = 240;

	Tatra T1; speed T1S;
	T1S.plain = 230; T1S.mountain = 250; T1S.desert = 240;

	K1.init("K1", K1S); T1.init("T1", T1S);

	K1.output(); cout << endl << endl; T1.output();

	cout << endl << endl;
	if (FrCreater(K1.getSpeed(1), T1.getSpeed(1)) == 1) cout << "Камаз быстрее Татры на равнинах!!!" << endl;
	else if (FrCreater(K1.getSpeed(1), T1.getSpeed(1)) == 0) cout << "Камаз и Татра равны по скорости на равнинах!!!" << endl;
	else cout << "Татра быстрее Kамаза на равнинах!!!" << endl;

	cout << endl;
	if (FrCreater(K1.getSpeed(2), T1.getSpeed(2)) == 1) cout << "Камаз быстрее Татры в горах!!!" << endl;
	else if (FrCreater(K1.getSpeed(2), T1.getSpeed(2)) == 0) cout << "Камаз и Татра равны по скорости в горах!!!" << endl;
	else cout << "Татра быстрее Kамаза в горах!!!" << endl;

	cout << endl;
	if (FrCreater(K1.getSpeed(3), T1.getSpeed(3)) == 1) cout << "Камаз быстрее Татры в пустыне!!!" << endl;
	else if (FrCreater(K1.getSpeed(3), T1.getSpeed(3)) == 0) cout << "Камаз и Татра равны по скорости в пустыне!!!" << endl;
	else cout << "Татра быстрее Kамаза в пустыне!!!" << endl;

	return 0;
}
