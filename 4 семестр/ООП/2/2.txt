#include <iostream>
#include <ctime>
#include <fstream>
#include <string>
#include "Windows.h"
#include <time.h>
#include <iomanip>

using namespace std;

struct info {
	str product_name;
	int wholesale_price = 0;
	int retail_price = 0;
	int count_product = 0;
};

class Nomenclature {
	info N;
public:
	Nomenclature(info temp) { N = temp; }
	void output() {
		cout << "| " << setw(17) << N.product_name << " | ";
		cout << setw(12) << N.wholesale_price << " | ";
		cout << setw(14) << N.retail_price << " | ";
		cout << setw(18) << N.count_product << " |" << endl;
	}
	int Estimation() { return ((N.retail_price - N.wholesale_price) * N.count_product); }
	str getName() { return N.product_name; }
};

int main() {
	SetConsoleCP(1251);
	SetConsoleOutputCP(1251);

	info temp;
	
	
	cin >> temp.product_name >> temp.wholesale_price;
	cin >> temp.retail_price >> temp.count_product;
	Nomenclature apple(temp);
	cin >> temp.product_name >> temp.wholesale_price;
	cin >> temp.retail_price >> temp.count_product;
	Nomenclature banana(temp);
	cin >> temp.product_name >> temp.wholesale_price;
	cin >> temp.retail_price >> temp.count_product;
	Nomenclature cherry(temp);

	cout << "__________________________________________________________________________" << endl;
	cout << "| Название продукта | Оптовая цена | Розничная цена | Кол - во на складе |" << endl;
	apple.output();
	banana.output();
	cherry.output();
	cout << "+------------------------------------------------------------------------+" << endl;
	cout << endl;
	cout << "Максимальная прибыль от " << apple.getName() << " составляет: " << apple.Estimation() << endl;
	cout << "Максимальная прибыль от " << banana.getName() << " составляет: " << banana.Estimation() << endl;
	cout << "Максимальная прибыль от " << cherry.getName() << " составляет: " << cherry.Estimation() << endl;

	return 0;
}
