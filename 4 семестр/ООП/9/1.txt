class air {
    static double p;
    static int countAllHit;
    static int countAllShoot;
    static int countShoot;
    static int countHit;
private:
    static int check(int& n) {
        if ((n * p) >= 1) {
            airShoot();
            return 1;
        }
        else {
            airMiss();
            return 0;
        }
    }
public:
    static void shoot(int n) {
        countAllShoot++;
        countShoot += n;
        countHit += check(n);
        countAllHit += countHit;
    }
    static void result() {
        green("За все время было выстрелено: "); red(to_string(countAllShoot)); green(" залпов\n");
        green("За все время попало: "); red(to_string(countAllHit)); green(" выстрелов\n\n");
        green("За последний залп было произведено: "); red(to_string(countShoot)); green(" выстрелов\n");
        green("За последний залп попали: "); red(to_string(countHit)); green(" выстрелов\n");
        countShoot = 0;
        countHit = 0;
    }
};

double air::p = 0.25;
int air::countAllHit = 0;
int air::countAllShoot = 0;
int air::countShoot = 0;
int air::countHit = 0;
