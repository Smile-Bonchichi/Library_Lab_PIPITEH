#include <iostream>
#include <ctime>
#include <fstream>
#include <string>
#include "Windows.h"
#include <time.h>
#include <iomanip>

#define ll long long
#define str string

using namespace std;

struct Plant {
	str name;
	int age = 0;
	str specialty;
	int salary = 0;
};

int main() {
	SetConsoleCP(1251);
	SetConsoleOutputCP(1251);

	fstream sin; 
	int n, countLocksmith = 0, countTurner = 0;
	sin.open("0\\Plant.txt");
	sin >> n;
	Plant* arr = new Plant[n];
	for (int i = 0; i < n; i++) {
		sin.ignore(); getline(sin, arr[i].name);
		sin >> arr[i].age;
		sin.ignore(); getline(sin, arr[i].specialty);
		sin >> arr[i].salary;
	}
	sin.close();

	for (int i = 0; i < n; i++) {
		if (arr[i].specialty == "Locksmith")
			countLocksmith++;
		else if (arr[i].specialty == "Turner")
			countTurner++;
	}
	cout << "_____________________________________________________" << endl;
	for (int i = 0; i < n; i++) {
		cout << "| " << i + 1 << " | " << setw(19) << arr[i].name << " | " << arr[i].age << " | " << setw(10) << arr[i].specialty << " | " << arr[i].salary << " |" << endl;
	}
	cout << "\n\nКоличество Токарей = " << countTurner << "\nКоличество Слесарей = " << countLocksmith << endl;
	delete[]arr;
	return 0;
}
