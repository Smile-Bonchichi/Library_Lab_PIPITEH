template <class T>
class Mass {
	int n;
	T* arr1;
	T* arr2;
public:
	Mass() { this->n = 0; arr1 = new T[n]; arr2 = new T[n]; }
	Mass(int count) { 
		this->n = count; 
		arr1 = new T[n];
		arr2 = new T[n];
		for (int i = 0; i < n; i++) {
			arr1[i] = rand() % 101 - 50;
			arr2[i] = rand() % 1001 - 500;
		}
	}
	T amount() {
		T sum = 0;
		for (int i = 0; i < n; i++)
			sum += (arr1[i] + arr2[i]);
		return sum;
	}
	void output() {
		red("#1\t\t #2\n");
		for (int i = 0; i < n; i++)
			cout << arr1[i] << "\t\t" << arr2[i] << endl;
	}
	~Mass() {
		delete[]arr1;
		delete[]arr2;
	}
};
