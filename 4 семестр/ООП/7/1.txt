template<class T>
T searchMax(T* arr, const int& n) {
	T max = arr[0];
	for (int i = 1; i < n; i++)
		if (arr[i] > max)
			max = arr[i];
	return max;
}

template<class T>
T* sort(T*& arr, const int& n){
	for (int j = 0; j < n - 1; j++)
		for (int i = 0; i < n - 1; i++)
			if (arr[i] > arr[i + 1])
				swap(arr[i], arr[i + 1]);
	return 0;
}
