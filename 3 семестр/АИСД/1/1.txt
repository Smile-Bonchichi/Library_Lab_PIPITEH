#include <iostream>
#include <string>
#include "Windows.h"

using namespace std;

int main() {
	SetConsoleCP(1251);
	SetConsoleOutputCP(1251);

	const int n = 10;
	int A[n], check = 0;
	double average, sum = 0;

	for (int i = 0; i < n; i++) {
		cin >> A[i];
		sum += A[i];
	}

	average = sum / n;
	cout << "\nСрередняя арифметическая = " << average << endl;

	for (int i = 0; i < n; i++) {
		if (A[i] > average) {
			check++;
			if (check > 0) {
				break;
			}
		}
	}

	if (check != 0) {
		cout << "\nЭлементы больше средней арифметической" << endl;
		for (int i = 0; i < n; i++) {
			if (A[i] > average) {
				cout << "A[" << i + 1 << "] = " << A[i] << endl;
			}
		}
	} else {
		cout << "\nЭлементов больше средней арифметической нет" << endl;
	}

	return 0;
}
